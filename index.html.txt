        // --- 核心安全工具：防止 null 導致畫面崩潰 ---
        function safeFixed(val, decimals = 2) {
            return (val !== null && val !== undefined && !isNaN(val)) ? Number(val).toFixed(decimals) : '--';
        }

        // --- 穩定的雙重 Proxy 抓取系統 ---
        async function fetchWithProxy(url) {
            // 提供多重 Proxy 備援機制，解決 API 被阻擋的問題
            const proxies =[
                `https://api.allorigins.win/raw?url=${encodeURIComponent(url)}`,
                `https://api.codetabs.com/v1/proxy?quest=${encodeURIComponent(url)}`
            ];
            
            for (let p of proxies) {
                try {
                    let res = await fetch(p, { cache: "no-store", timeout: 8000 });
                    if (!res.ok) continue;
                    let text = await res.text();
                    // 防呆：如果 Proxy 回傳了 Yahoo 的防爬蟲 HTML 網頁，跳過不用
                    if (text.includes('html>') || text.includes('Will be right back')) continue;
                    
                    let json = JSON.parse(text);
                    if (json) return json;
                } catch (e) {
                    console.warn(`Proxy failed: ${p}`);
                }
            }
            return null; // 全數失敗
        }

        // --- 核心演算法：計算真實 RSI、MACD、布林通道 ---
        function calculateEMA(data, period) {
            let ema = new Array(data.length).fill(null);
            let multiplier = 2 / (period + 1);
            let sum = 0, count = 0;
            for(let i=0; i<data.length; i++) {
                if(data[i] !== null && !isNaN(data[i])) {
                    if(count < period) { sum += data[i]; count++; if(count===period) ema[i] = sum/period; } 
                    else { ema[i] = (data[i] - ema[i-1]) * multiplier + ema[i-1]; }
                }
            }
            return ema;
        }

        function calcAdvancedIndicators(data) {
            let closes = data.closes;
            if(closes.length === 0) return;

            // MA
            data.ma5 = calculateEMA(closes, 5);
            data.ma10 = calculateEMA(closes, 10);
            data.ma20 = calculateEMA(closes, 20);
            data.ma60 = calculateEMA(closes, 60);

            // RSI(14)
            let rsi = new Array(closes.length).fill(null);
            let period = 14;
            if(closes.length >= period) {
                let gains = 0, losses = 0;
                for(let i=1; i<=period; i++) {
                    let diff = closes[i] - closes[i-1];
                    if(diff>0) gains+=diff; else losses-=diff;
                }
                let avgGain = gains/period, avgLoss = losses/period;
                rsi[period] = 100 - (100 / (1 + avgGain/(avgLoss===0?0.001:avgLoss)));
                for(let i=period+1; i<closes.length; i++) {
                    let diff = closes[i] - closes[i-1];
                    let gain = diff>0 ? diff : 0;
                    let loss = diff<0 ? -diff : 0;
                    avgGain = (avgGain*(period-1)+gain)/period;
                    avgLoss = (avgLoss*(period-1)+loss)/period;
                    rsi[i] = 100 - (100 / (1 + avgGain/(avgLoss===0?0.001:avgLoss)));
                }
            }
            data.rsi = rsi;

            // MACD (12, 26, 9)
            let ema12 = calculateEMA(closes, 12);
            let ema26 = calculateEMA(closes, 26);
            let macd = new Array(closes.length).fill(null);
            for(let i=0; i<closes.length; i++) {
                if(ema12[i]!==null && ema26[i]!==null) macd[i] = ema12[i]-ema26[i];
            }
            let signal = calculateEMA(macd, 9);
            data.macd = macd;
            data.macdSignal = signal;

            // Bollinger Bands (20, 2)
            let upperBB = new Array(closes.length).fill(null);
            let lowerBB = new Array(closes.length).fill(null);
            for(let i=19; i<closes.length; i++) {
                let slice = closes.slice(i-19, i+1).filter(val => val !== null);
                if(slice.length === 20) {
                    let mean = slice.reduce((a,b)=>a+b,0)/20;
                    let variance = slice.reduce((a,b)=>a+Math.pow(b-mean,2),0)/20;
                    let stddev = Math.sqrt(variance);
                    upperBB[i] = mean + 2*stddev;
                    lowerBB[i] = mean - 2*stddev;
                }
            }
            data.upperBB = upperBB;
            data.lowerBB = lowerBB;
        }

        // --- 獲取 100% 真實數據 ---
        async function fetchRealData(stock, days) {
            let warnEl = document.getElementById('data-warning');
            let titleEl = document.getElementById('chart-title');
            warnEl.classList.remove('hidden');
            warnEl.innerText = "⏳ 正在突破防爬蟲限制抓取真實數據...";
            titleEl.innerText = `連線抓取中...`;

            // 自動判斷上市 (.TW) 或 上櫃 (.TWO)
            let finalSymbol = stock.match(/^[a-zA-Z]+$/) ? stock : stock + ".TW"; // 美股或台股
            let quoteUrl = `https://query2.finance.yahoo.com/v7/finance/quote?symbols=${stock}.TW,${stock}.TWO,${stock}`;
            let quoteData = await fetchWithProxy(quoteUrl);
            let quote = null;
            if (quoteData && quoteData.quoteResponse && quoteData.quoteResponse.result.length > 0) {
                quote = quoteData.quoteResponse.result[0];
                finalSymbol = quote.symbol; 
            }

            // 抓取真實歷史股價
            let chartUrl = `https://query2.finance.yahoo.com/v8/finance/chart/${finalSymbol}?range=2y&interval=1d`;
            let chartData = await fetchWithProxy(chartUrl);
            
            if (!chartData || !chartData.chart || !chartData.chart.result) {
                titleEl.innerText = `載入失敗！`;
                warnEl.innerHTML = `❌ 無法取得 <b>${stock}</b> 數據。原因可能為：代碼錯誤、未上市櫃、或 Proxy 被瀏覽器阻擋。`;
                return;
            }

            let result = chartData.chart.result[0];
            let timestamps = result.timestamp;
            let q = result.indicators.quote[0];
            
            let data = { dates: [], opens: [], highs:[], lows:[], closes:[], volumes:[], quote: quote, symbol: finalSymbol };
            for(let i = 0; i < timestamps.length; i++) {
                // 過濾掉盤中暫停交易產生的 null
                if(q.close[i] !== null && q.close[i] !== undefined) {
                    let d = new Date(timestamps[i] * 1000);
                    // 格式化為真實日期 YYYY-MM-DD
                    let dateStr = d.toLocaleDateString('sv-SE', { timeZone: 'Asia/Taipei' }); 
                    data.dates.push(dateStr);
                    data.opens.push(q.open[i]); data.highs.push(q.high[i]);
                    data.lows.push(q.low[i]); data.closes.push(q.close[i]);
                    data.volumes.push(q.volume[i] || 0);
                }
            }

            calcAdvancedIndicators(data);

            // 抓取真實 RSS 新聞 (轉換為 JSON)
            let rssUrl = `https://feeds.finance.yahoo.com/rss/2.0/headline?s=${finalSymbol}`;
            let newsUrl = `https://api.rss2json.com/v1/api.json?rss_url=${encodeURIComponent(rssUrl)}`;
            try {
                let newsRes = await fetch(newsUrl);
                if(newsRes.ok) {
                    let newsData = await newsRes.json();
                    data.news = newsData.items || [];
                } else data.news =[];
            } catch(e) { data.news =[]; }

            warnEl.classList.add('hidden');
            stockDataCache[stock] = data;

            renderTabs();
            updateTopPriceBar();
            updateChart();
            updatePanelsPreview(stock, data);
            runAdvancedAIStrategies(stock);
        }

        function updateTopPriceBar() {
            let data = stockDataCache[activeStock];
            if(!data || data.closes.length === 0) return;

            let name = data.quote ? data.quote.shortName : activeStock;
            document.getElementById('chart-title').innerText = `${name}`;
            
            let currentClose = data.closes[data.closes.length - 1];
            let prevClose = data.closes[data.closes.length - 2] || currentClose;
            let diff = currentClose - prevClose;
            let pct = prevClose ? (diff / prevClose) * 100 : 0;
            
            let priceEl = document.getElementById('current-price');
            let changeEl = document.getElementById('price-change');
            
            priceEl.innerText = safeFixed(currentClose, 2);
            if(diff > 0) {
                priceEl.className = "text-3xl font-bold text-red-500 tick-up";
                changeEl.className = "text-lg font-semibold text-red-500"; changeEl.innerText = `▲ ${safeFixed(diff)} (+${safeFixed(pct)}%)`;
            } else if(diff < 0) {
                priceEl.className = "text-3xl font-bold text-green-500 tick-down";
                changeEl.className = "text-lg font-semibold text-green-500"; changeEl.innerText = `▼ ${safeFixed(Math.abs(diff))} (${safeFixed(pct)}%)`;
            } else {
                priceEl.className = "text-3xl font-bold text-gray-300"; changeEl.className = "text-lg font-semibold text-gray-400"; changeEl.innerText = `－ 0.00 (0.00%)`;
            }
        }

        // --- K線橫座標顯示「真實日期」 ---
        function updateChart() {
            let data = stockDataCache[activeStock];
            if(!data || data.dates.length === 0) return;

            // 依天數裁切
            let bars = currentRange === 30 ? 22 : currentRange === 90 ? 65 : currentRange === 180 ? 130 : 250;
            let sliceStart = Math.max(0, data.dates.length - bars);
            
            let sDates = data.dates.slice(sliceStart);
            let traces =[{
                x: sDates, open: data.opens.slice(sliceStart), high: data.highs.slice(sliceStart), 
                low: data.lows.slice(sliceStart), close: data.closes.slice(sliceStart),
                type: 'candlestick', name: 'K線',
                increasing: {line: {color: '#ef4444'}}, decreasing: {line: {color: '#22c55e'}}
            }];

            if(document.getElementById('ma5').checked) traces.push({ x: sDates, y: data.ma5.slice(sliceStart), type: 'scatter', mode: 'lines', name: 'MA5', line: {color: '#facc15', width: 1.5} });
            if(document.getElementById('ma10').checked) traces.push({ x: sDates, y: data.ma10.slice(sliceStart), type: 'scatter', mode: 'lines', name: 'MA10', line: {color: '#fb923c', width: 1.5} });
            if(document.getElementById('ma20').checked) traces.push({ x: sDates, y: data.ma20.slice(sliceStart), type: 'scatter', mode: 'lines', name: 'MA20', line: {color: '#c084fc', width: 1.5} });
            if(document.getElementById('ma60').checked) traces.push({ x: sDates, y: data.ma60.slice(sliceStart), type: 'scatter', mode: 'lines', name: 'MA60', line: {color: '#60a5fa', width: 1.5} });

            let layout = {
                paper_bgcolor: 'rgba(0,0,0,0)', plot_bgcolor: 'rgba(0,0,0,0)', font: {color: '#d1d5db'},
                margin: {t: 10, b: 30, l: 40, r: 10},

        }

        // --- 綜合 AI 多模型交易策略 (全面擴充) ---
        function runAdvancedAIStrategies(stock) {
            let data = stockDataCache[stock];
            if(!data || data.closes.length < 20) {
                document.getElementById('sig-strategy-name').innerText = "資料不足無法評估";
                return;
            }

            let lastIdx = data.closes.length-1;
            let lastClose = data.closes[lastIdx];
            let rsi = data.rsi[lastIdx];
            let macd = data.macd[lastIdx];
            let macdSig = data.macdSignal[lastIdx];
            let ma5 = data.ma5[lastIdx];
            let ma20 = data.ma20[lastIdx];
            let upperBB = data.upperBB[lastIdx];
            let lowerBB = data.lowerBB[lastIdx];

            let strategies =[];

            // 模型 1: RSI 超賣反彈策略
            if (rsi !== null && rsi < 35 && lastClose >= lowerBB) {
                strategies.push({ name: "【RSI 極度超賣反彈】", score: 85 + (35-rsi), dir: "做多 🔴", entry: safeFixed(lastClose), exit: safeFixed(lowerBB), reason: `技術指標顯示 RSI 已跌至 ${safeFixed(rsi,1)} 的極度超賣區，且股價在布林下軌 (${safeFixed(lowerBB)}) 找到支撐，具備強烈短線反彈動能。` });
            }
            // 模型 2: 布林通道強勢突破策略
            if (upperBB !== null && lastClose > upperBB && ma5 > ma20) {
                strategies.push({ name: "【布林通道強勢突破】", score: 90, dir: "做多 🔴", entry: safeFixed(lastClose), exit: safeFixed(ma5), reason: `股價強勢帶量突破布林上軌 (${safeFixed(upperBB)})，均線呈現多頭排列，代表波段多頭攻勢啟動。防守停損點建議設於 5 日線。` });
            }
            // 模型 3: MACD 與 均線多頭順勢策略
            if (ma5 !== null && ma20 !== null && macd !== null && ma5 > ma20 && macd > macdSig && rsi >= 45 && rsi <= 65) {
                strategies.push({ name: "【MACD 均線多頭順勢】", score: 80, dir: "做多 🔴", entry: `拉回 ${safeFixed(ma5)}`, exit: safeFixed(ma20), reason: `短期 MA5 大於 MA20，且 MACD 柱狀體維持多方領域。趨勢穩定向上，建議於 5 日線附近伺機順勢佈局。` });
            }
            // 模型 4: 高檔超買乖離回檔策略
            if (rsi !== null && rsi > 70 && lastClose < ma5) {
                strategies.push({ name: "【高檔超買乖離回檔】", score: 82, dir: "做空 🟢", entry: safeFixed(lastClose), exit: safeFixed(data.highs[lastIdx]), reason: `RSI 高達 ${safeFixed(rsi,1)} 進入過熱超買區，且股價已率先跌破 5 日均線，顯示多頭力竭，有回測月線壓力的極大可能。` });
            }
            // 模型 5: 均線空頭規避策略
            if (ma5 !== null && ma20 !== null && ma5 < ma20 && macd < macdSig) {
                strategies.push({ name: "【均線空頭規避策略】", score: 88, dir: "觀望/做空 🟢", entry: `反彈 ${safeFixed(data.ma10[lastIdx])}`, exit: safeFixed(ma20), reason: `月線下彎且 MACD 位於零軸之下，屬於空方控盤格局。切勿輕易接刀，建議空手觀望或於反彈至 10 日線時偏空操作。` });
            }

            // 綜合評估：選出分數最高的最優策略
            let bestStrategy = null;
            if (strategies.length > 0) {
                strategies.sort((a,b) => b.score - a.score);
                bestStrategy = strategies[0];
            } else {
                // 模型 6: 無趨勢震盪策略
                bestStrategy = { name: "【量縮無趨勢震盪區間】", score: 50, dir: "觀望 🟡", entry: "等待帶量表態", exit: "不適用", reason: `目前各項真實技術指標（MACD、均線糾結）皆無明顯單邊方向，處於無趨勢的區間震盪整理，貿然進場勝率低，建議空手等待明確表態。` };
            }

            document.getElementById('sig-strategy-name').innerText = bestStrategy.name;
            document.getElementById('sig-direction').innerHTML = `<span class="${bestStrategy.dir.includes('做多')?'text-red-500':(bestStrategy.dir.includes('做空')?'text-green-500':'text-yellow-500')}">${bestStrategy.dir}</span>`;
            document.getElementById('sig-entry').innerText = bestStrategy.entry;
            document.getElementById('sig-exit').innerText = bestStrategy.exit;
            document.getElementById('sig-reason').innerText = bestStrategy.reason;
        }
